# 8. String to Integer
## Time Failed Solution
```python
import re
class Solution:
    def myAtoi(self, s: str) -> int:
        t=1
        if "-" in s:
            t=-1
            
        if bool(re.search('[a-z]', s))==True:
            m = re.search("\d", s)
            mm = re.search("[a-z]", s)
            if mm.start()<m.start():
                return 0
        
        for i in range(len(s)):
            if s[i].isalpha()==True:
                if i==0:
                    return 0
                s = s[:i]
                break
                
        s =re.findall(r"\d+\.?\d*",s)    
        s = "".join(s)
        if '.' in s:
            s = t*float(s)
        else:
            s = t*int(s)
        
        if s in range(-2**31,2**31):
            return s
        else:
            return t*2**31
```
